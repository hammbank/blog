（3）开发代码 

package org.example.flink_demo002

import java.sql.{Connection, DriverManager, PreparedStatement}
import org.apache.flink.configuration.Configuration
import org.apache.flink.streaming.api.functions.source.{RichSourceFunction, SourceFunction}
import org.apache.flink.streaming.api.scala.{DataStream, StreamExecutionEnvironment}
import org.apache.log4j.{Level, Logger}

object FlinkStreamSourceFromMySQL {
  Logger.getLogger("org").setLevel(Level.ERROR)
  def main(args: Array[String]): Unit = {
    //todo: 构建流处理环境
    val environment: StreamExecutionEnvironment = StreamExecutionEnvironment.getExecutionEnvironment
    import org.apache.flink.api.scala._
    //todo: 添加自定义source    获取数据源的source task 个数为1，单并行度
    val dataStream: DataStream[User] = environment.addSource(new MySQLSource)
    //todo: 打印处理
    dataStream.setParallelism(1).print()
    //todo： 启动
    environment.execute()
  }
}

//todo: 自定义单并行度source
class MySQLSource extends RichSourceFunction[User] {

  private var connection: Connection = null
  private var ps: PreparedStatement = null

  override def open(parameters: Configuration): Unit = {
    val url = "jdbc:mysql://node03:3306/my_flink_testdb?useSSL=false&serverTimezone=Asia/Shanghai&useUnicode=true&characterEncoding=utf-8"
    val username = "root"
//    Class.forName("com.mysql.jdbc.Driver")
    connection = DriverManager.getConnection(url, username, "123456")
    val sql = "select * from user"
    ps = connection.prepareStatement(sql)
  }

  override def close(): Unit = {
    if (connection != null) {
      connection.close()
    }
    if (ps != null) {
      ps.close()
    }
  }

  override def run(sourceContext: SourceFunction.SourceContext[User]): Unit = {
    //todo: 执行SQL并收集数据
    val rs = ps.executeQuery()
    while (rs.next()) {
      val user = User(rs.getInt("id"), rs.getString("name"), rs.getInt("age"))
      sourceContext.collect(user)
    }
    rs.close()
  }

  override def cancel(): Unit = {
  }
}

case class User(id: Int, name: String, age: Int)

完整的scala代码如下： 

package org.example.flink_demo002

import org.apache.flink.streaming.api.scala.{DataStream, StreamExecutionEnvironment}


//todo: 通过scala开发flink流处理作业
object TestParallelism {

  def main(args: Array[String]): Unit = {

    //todo: 1、构建流处理的环境
    val env: StreamExecutionEnvironment = StreamExecutionEnvironment.createLocalEnvironmentWithWebUI()
    //todo: 2、从socket获取数据
    val sourceStream: DataStream[String] = env.socketTextStream("node01", 9999)

    //导入隐式转换的包
    import org.apache.flink.api.scala._

    //todo: 3、对数据进行处理
    val result: DataStream[(String, Int)] = sourceStream
      .flatMap(_.split(" ")).setParallelism(5) //按照空格切分
      .map((_, 1)).setParallelism(5) //每个单词计为1
      .keyBy(_._1) //按元组中的第1个元素进行分组，即按单词分组
      .sum(1).setParallelism(4) //按照下标为1累加相同单词出现的次数

    //todo： 4、打印输出，sink
    result.print().setParallelism(4)

    //todo: 5、开启任务
    env.execute("Job_FlinkWordCountStream")

  }

}
