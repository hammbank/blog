（2）开发代码

src/main/scala/com/kkb/FlinkWordCountStreamDemo/FlinkWordCountStreamDemo.scala: 

package com.kkb.FlinkWordCountStreamDemo

import org.apache.flink.streaming.api.scala.{DataStream, StreamExecutionEnvironment}

//todo: 通过scala开发flink流处理作业
object FlinkWordCountStreamDemo {

  def main(args: Array[String]): Unit = {

    //todo: 1、构建流处理的环境
    val env: StreamExecutionEnvironment = StreamExecutionEnvironment.getExecutionEnvironment

    //todo: 2、从socket获取数据
    val sourceStream: DataStream[String] = env.socketTextStream("node01", 9999)

    //导入隐式转换的包
    import org.apache.flink.api.scala._

    //todo: 3、对数据进行处理
    val result: DataStream[(String, Int)] = sourceStream
      .flatMap(_.split(" ")) //按照空格切分
      .map((_, 1)) //每个单词计为1
      .keyBy(_._1) //按元组中的第1个元素进行分组，即按单词分组
      .sum(1) //按照下标为1累加相同单词出现的次数

    //todo： 4、打印输出，sink
    result.print()

    //todo: 5、开启任务
    env.execute("Job_FlinkWordCountStream")

  }

}

（3）程序打成jar包，提交到yarn中运行 

mvn clean package

bin/flink run -m yarn-cluster -yjm 1024 -ytm 1024 -c com.kkb.FlinkWordCountStreamDemo.FlinkWordCountStreamDemo ~/original-flink_demo01-1.0-SNAPSHOT.jar
